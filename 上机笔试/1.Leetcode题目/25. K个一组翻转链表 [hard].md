### 25. K个一组翻转链表 [hard]

递归翻转，前翻转前K个，然后再翻转后面的。

```python
class Solution:
    def reverseKGroup(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        def reverse(head): ##reverse a list
            curr = head
            prev = None
            while curr:
                curr_next = curr.next
                curr.next = prev
                prev = curr
                curr = curr_next
            return prev

        tmp = head
        for i in range(k-1):
            if not tmp: return head ###不够翻转，直接返回
            tmp = tmp.next
        if not tmp:
            return head
        
        right_list = tmp.next
        tmp.next = None ##【易错】不要忘记切断
        right_list = self.reverseKGroup(right_list,k)
        new_head = reverse(head)
        tmp = new_head
        while tmp and tmp.next:
            tmp = tmp.next
        tmp.next = right_list ##再次链接
        return new_head
```
